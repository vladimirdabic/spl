broj a = 20;
bool b = true;

// Niska i niz su istog tipa
niska msg = "Hello World";
niz msg2 = "Hello World";

// Svaka tip promenljive ima 'default value'
// Broj - 0
// Niska/Niz - ""
// Bool - false
broj num1;

// Tip: promenljiva
// Ovaj tip prima bilo koju vrednost
promenljiva x = "Hello";
